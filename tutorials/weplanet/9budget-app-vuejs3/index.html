예산 앱을 탐색하기에는 수동으로 URL을 조정하는 것보다 예산 목록 보기 페이지를 넣는 것이 간단할 것이다. 또한 계정을 리팩토링할 때 빠뜨린 예산 네이밍에서 View를 삭제 할 것이다.

src/app/budgets/routes.js

import * as components from './components';

export default [
  {
    path: '/budgets',
    component: components.BudgetsList,
    name: 'budgetsList'
  },
  {
    path: '/budgets/create',
    component: components.CreateUpdateBudget,
    name: 'createBudget'
  },
  {
    path: '/budgets/:budgetId/update',
    component: components.CreateUpdateBudget,
    name: 'updateBudget'
  }
];

src/app/budgets/components/index.js

export { default as BudgetsList } from './BudgetsList';
export { default as CreateUpdateBudget } from './CreateUpdateBudget';

export default {
  name: 'budgets-list',

  mounted () {
    this.loadBudgets();
  },

  methods: {
    ...mapActions([
      'loadBudgets'
    ])
  },

  computed: {
    ...mapState({
      'budgets': state => state.budgets.budgets
